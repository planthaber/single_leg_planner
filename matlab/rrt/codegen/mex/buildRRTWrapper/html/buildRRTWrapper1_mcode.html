1%BUILDRRTWRAPPER This function acts as a wrapper for the buildRRT function.
  2%Code generation for the singleLegPlanner is performed using this function
  3%as an entry point.
  4%  
  5%Inputs:
  6%-nInitCartesianB: the
  7%-nGoalCartesianB:
  8%-jointLimits:
  9%-bodyHeight:
 10%-U:
 11%-dt:
 12%-Dt:
 13%-kC:
 14%-threshold:
 15%-legNum:
 16%
 17%Outputs:
 18%-T:
 19%-pathC:
 20%-pathJ:
 21%-success:
 22%
 23%buildRRTWrapper.m
 24%author: wreid
 25%date: 20150502
 26
 27function [T,pathC,pathJ,success] = buildRRTWrapper(nInitCartesianB,nGoalCartesianB,phiInit,omegaInit,jointLimits,bodyHeight,U,dt,Dt,kC,threshold,legNum,uBDot)
 28
 29    persistent NUM_NODES 
 30    persistent NODE_SIZE 
 31    persistent U_SIZE 
 32    persistent ankleThreshold 
 33    persistent exhaustive 
 34    persistent goalSeedFreq 
 35    persistent cartesianLimits
 36    persistent HGAINS
 37    
 38    %if isempty(NUM_NODES)
 39    %    NUM_NODES = int32(1000);
 40    %end
 41    if isempty(NODE_SIZE)
 42        NODE_SIZE = int32(13);
 43    end
 44    if isempty(U_SIZE)
 45        U_SIZE = int32(5);
 46    end
 47    if isempty(ankleThreshold)
 48        ankleThreshold = pi/8;
 49    end
 50    if isempty(exhaustive)
 51        exhaustive = false;
 52    end
 53    if isempty(goalSeedFreq)
 54        goalSeedFreq = int32(20);
 55    end
 56    if isempty(cartesianLimits)
 57        cartesianLimits = [-0.2930   -1.1326   -0.6710   -0.7546];
 58    end
 59    if isemtpy(NUM_NODES)
 60        NUM_NODES = 1000;
 61    end
 62    if isempty(HGAINS)
 63        HGAINS = [1 0 0.5];
 64    end
 65    
 66    panHeight  = getPanHeight(bodyHeight,kC);
 67
 68    %Transform the nInitCartesianB and nGoalCartesianB variables from the body coordinate frame
 69    %to the pan coordinate frame.
 70    TP2B = trP2B(kC,legNum);
 71    TB2P = trInv(TP2B); %inv(TP2B);%
 72    uInitP = TB2P(1:3,1:3)*nInitCartesianB(1:3)' + TB2P(1:3,4);
 73    uDotInitP = TB2P(1:3,1:3)*nInitCartesianB(4:6)';
 74    uGoalP = TB2P(1:3,1:3)*nGoalCartesianB(1:3)' + TB2P(1:3,4);
 75    uDotGoalP = TB2P(1:3,1:3)*nGoalCartesianB(4:6)';
 76    
 77    %Transform the Cartesian goal and final positions in the pan coordinate
 78    %frame to the joint space.
 79    qInit = sherpaTTIK(uInitP,kC,jointLimits);
 80    qGoal = sherpaTTIK(uGoalP,kC,jointLimits);
 81    qDotInit = sherpaTTIKVel(uDotInitP',qInit',kC);
 82    qDotGoal = sherpaTTIKVel(uDotGoalP',qGoal',kC);
 83    
 84    nInitJoint = [qInit phiInit 0 qDotInit' 0 omegaInit];
 85    nGoalJoint = [qGoal 0 0 qDotGoal' 0 0];
 86    
 87    %Check that the initial and final positions are valid. If they are not
 88    %return failure and an empty path.
 89    if (validJointState(nInitJoint,jointLimits) && validJointState(nGoalJoint,jointLimits))
 90        success = true;
 91        %Run buildRRT.
 92        nInit = [1 0 0 nInitJoint];
 93        nGoal = [0 0 0 nGoalJoint];
 94        [T,pathJ] = buildRRT(nInit,nGoal,NUM_NODES,jointLimits,cartesianLimits,panHeight,HGAINS,NODE_SIZE,U,U_SIZE,dt,Dt,kC,ankleThreshold,exhaustive,threshold,goalSeedFreq,uBDot,legNum);
 95        %Transform path back to the Cartesian space.
 96        pathC = transformPath(pathJ,kC,TP2B);
 97    else
 98        success = false;
 99        pathC = [];
100        pathJ = [];
101        T = [];
102    end 
103
104end

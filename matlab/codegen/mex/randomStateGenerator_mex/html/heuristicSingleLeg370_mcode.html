 5function d = heuristicSingleLeg(xA,xB,HGAINS,jointLimits,kC)
 6%heuristic Calculates the distance between states x1 and x2.
 7
 8    alphaA = xA(4); 
 9    betaA = xA(5);
10    gammaA = xA(6);
11    phiA = xA(7);
12    epsilonA = xA(8);
13    
14    alphaB = xB(4);
15    betaB = xB(5);
16    gammaB = xB(6);
17    phiB = xB(7);
18    epsilonB = xB(8);
19    
20    alphaDotA = xA(9);
21    betaDotA = xA(10);
22    gammaDotA = xA(11);
23    phiDotA = xA(12);
24    omegaA = xA(13);
25    
26    alphaDotB = xB(9);
27    betaDotB = xB(10);
28    gammaDotB = xB(11);
29    phiDotB = xB(12);
30    omegaB = xB(13);
31    
32    %Calculate the distance between angular positions.
33    xStarMin = legRadius(jointLimits(1,2),jointLimits(1,3),kC);
34    xStarMax = legRadius(jointLimits(2,2),jointLimits(2,3),kC);
35    
36    dxStarMax = xStarMax-xStarMin;
37    dAlphaMax = angDiff(jointLimits(1,1),jointLimits(1,2));
38    
39    dPosMax = posMetric(xStarMin,dxStarMax,dAlphaMax);
40    
41    xStarA = legRadius(betaA,gammaA,kC);
42    xStarB = legRadius(betaB,gammaB,kC);
43    
44    dxStar = xStarB-xStarA;
45    dAlpha = angDiff(alphaA,alphaB);
46    
47    dPos = sqrt(dxStar^2+xStarA^2*dAlpha^2);
48    
49    dPosNorm = dPos/dPosMax;
50    
51    uA = sherpaTTFK([alphaA betaA gammaA],kC);
52    uB = sherpaTTFK([alphaB betaB gammaB],kC);
53    
54    d = norm(uB-uA);
55    
56    
57    %dVel = (alphaDotB - alphaDotA) + (betaDotB - betaDotA) + (gammaDotB - gammaDotA);
58    %dVelNorm = jointLimits(2,6) - jointLimits(1,6) + jointLimits(2,7) - jointLimits(1,7) + jointLimits(2,8) - jointLimits(1,8);
59    
60%    uA = sherpaTTFK(xA(4:6),kC);
61%    uB = sherpaTTFK(xB(4:6),kC);
62    %dPos = norm(uA-uB);
63    
64    %Calculate the total distance.
65    %d = HGAINS(1)*dPosNorm;%+HGAINS(2)*dVelNorm; 
66    
67end

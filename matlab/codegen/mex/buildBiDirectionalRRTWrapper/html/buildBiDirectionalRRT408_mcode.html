  5function [T1,T2,pathJ,pathC] = buildBiDirectionalRRT(nInit,nGoal,NUM_NODES,jointLimits,cartesianLimits,panHeight,HGAINS,NODE_SIZE,U,U_SIZE,dt,Dt,kC,ankleThreshold,exhaustive,threshold,goalSeedFreq,uBDot,legNum,TP2B)
  6%buildRRT Icrementally builds a rapidly exploring random tree.
  7%   An RRT is build by incrementally selecting a random state from the
  8%   available state space as defined by the MIN and MAX vectors. The tree is
  9%   started at xInit and is extended until the number of maximum nodes, K has
 10%   been reached. A path is selected if the goal region as defined by xGoal
 11%   has been reached by the RRT.
 12
 13    %Constant Declaration                                                      
 14    transitionArrayLength = (round(Dt/dt)+1)*10;    
 15    
 16    T1 = zeros(round(NUM_NODES/2),NODE_SIZE+transitionArrayLength);     
 17    T2 = zeros(round(NUM_NODES/2),NODE_SIZE+transitionArrayLength);
 18    
 19    T1(1,:) = [nInit zeros(1,transitionArrayLength)];           
 20    T2(1,:) = [nGoal zeros(1,transitionArrayLength)];
 21    
 22    nMid1 = [];
 23    nMid2 = [];
 24    dMin = 100;
 25    
 26    nodeIDCount1 = 1;
 27    nodeIDCount2 = 1;
 28    
 29    path = [];
 30    pathCount = 1;
 31    
 32    pathLengthMin = 100;
 33    pathCMin = [];
 34    pathJ = [];
 35    pathC = [];
 36
 37    for i = 3:NUM_NODES
 38        
 39        [T1,~,~] = rrtLoop(T1,jointLimits,cartesianLimits,kC,panHeight,U,Dt,dt,NODE_SIZE,U_SIZE,HGAINS,ankleThreshold,nodeIDCount1,nGoal,goalSeedFreq,uBDot,legNum);
 40        nodeIDCount1 = nodeIDCount1 + 1;
 41
 42        %Swap the trees.
 43        TTemp = T1;
 44        T1 = T2;
 45        T2 = TTemp;
 46        
 47        %Swap the trees.
 48        nodeIDCountTemp = nodeIDCount1;
 49        nodeIDCount1 = nodeIDCount2;
 50        nodeIDCount2 = nodeIDCountTemp;
 51        
 52    end
 53    
 54    [T1H,~] = size(T1);
 55    [T2H,~] = size(T2);
 56    
 57    for i = 1:T1H
 58        
 59        nMid1 = T1(i,1:NODE_SIZE);
 60        [nMid2,~,d] = nearestNeighbour(nMid1,T2,[0.9 0.1 0],jointLimits,kC,T2H,NODE_SIZE);
 61        
 62        if d < 0.04
 63            pathT1 = traceBranch(T1,nMid1,NODE_SIZE);
 64            pathT2 = traceBranch(T2,nMid2,NODE_SIZE);
 65            if T1(1,4) == nInit(1,4) && T1(1,5) == nInit(1,5) && T1(1,6) == nInit(1,6)
 66                path = [pathT1;flipud(pathT2)];
 67            else
 68                path = [pathT2;flipud(pathT1)];
 69            end
 70            
 71            [pathH,~] = size(path);
 72            t = dt*(1:pathH)';
 73            path = [t path];
 74
 75            pathC = transformPath(path,kC,TP2B);
 76            pathLength = pathC(end,2);
 77
 78            if (pathLength < pathLengthMin)
 79                pathLengthMin = pathLength;
 80                pathCMin = pathC;
 81                pathJ = path;
 82            end
 83        end
 84    end
 85    pathC = pathCMin;
 86end

129function pathC = transformPath(pathJ,kC,TP2B)
130    [pathH,~] = size(pathJ);
131    pathC = zeros(pathH,9);
132    dist2Go = 0;
133    for i = 1:pathH
134        uP = sherpaTTFK(pathJ(i,2:4),kC);
135        uPDot = sherpaTTFKVel(pathJ(i,7:9),pathJ(i,2:4),kC);
136        uB = TP2B(1:3,1:3)*uP' + TP2B(1:3,4);
137        uBDot = TP2B(1:3,1:3)*uPDot;
138        if i ~= 1
139            dist2Go = dist2Go + norm(uB'-pathC(i-1,3:5));
140        end
141        pathC(i,:) = [pathJ(i,1) dist2Go uB' uBDot' false];
142    end
143end

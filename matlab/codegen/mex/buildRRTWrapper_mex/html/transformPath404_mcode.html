 96function pathC = transformPath(pathJ,kC,TP2B)
 97    [pathH,~] = size(pathJ);
 98    pathC = zeros(pathH,9);
 99    dist2Go = 0;
100    for i = 1:pathH
101        uP = sherpaTTFK(pathJ(i,2:4),kC);
102        uPDot = sherpaTTFKVel(pathJ(i,7:9),pathJ(i,2:4),kC);
103        uB = TP2B(1:3,1:3)*uP' + TP2B(1:3,4);
104        uBDot = TP2B(1:3,1:3)*uPDot;
105        if i ~= 1
106            dist2Go = dist2Go + norm(uB'-pathC(i-1,3:5));
107        end
108        pathC(i,:) = [pathJ(i,1) dist2Go uB' uBDot' false];
109    end
110end

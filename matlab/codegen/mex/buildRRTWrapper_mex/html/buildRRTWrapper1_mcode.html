1%BUILDRRTWRAPPER This function acts as a wrapper for the buildRRT function.
  2%Code generation for the singleLegPlanner is performed using this function
  3%as an entry point.
  4%  
  5%Inputs:
  6%-nInitCartesianB: the
  7%-nGoalCartesianB:
  8%-jointLimits:
  9%-bodyHeight:
 10%-kC:
 11%-legNum:
 12%-uBDot:
 13%
 14%Outputs:
 15%-T:
 16%-pathC:
 17%-pathJ:
 18%-success:
 19%
 20%buildRRTWrapper.m
 21%author: wreid
 22%date: 20150502
 23
 24function [T,pathC,pathJ,success] = buildRRTWrapper(nInitCartesianB,nGoalCartesianB,phiInit,omegaInit,jointLimits,bodyHeight,kC,legNum,uBDot)
 25
 26    NUM_NODES = int32(1500);
 27    NODE_SIZE = int32(13);
 28    U_SIZE = int32(5);
 29    ankleThreshold = 5*pi/180;
 30    exhaustive = false;
 31    goalSeedFreq = int32(20);
 32    cartesianLimits = [-0.2930   -1.1326   -0.6710   -0.7546];
 33    dt = 0.1;
 34    Dt = 0.7;
 35    HGAINS = [1 0 0.5];
 36    threshold = 0.005;
 37    stepAccRatio = 14;
 38    eta = Dt/stepAccRatio;
 39    U = eta*[1 0; -1 0; 0 1; 0 -1; 0 0];     
 40    panHeight  = getPanHeight(bodyHeight,kC);
 41
 42    %Transform the nInitCartesianB and nGoalCartesianB variables from the body coordinate frame
 43    %to the pan coordinate frame.
 44    TP2B = trP2B(kC,legNum);
 45    TB2P = trInv(TP2B); %inv(TP2B);%
 46    uInitP = TB2P(1:3,1:3)*nInitCartesianB(1:3)' + TB2P(1:3,4);
 47    uDotInitP = TB2P(1:3,1:3)*nInitCartesianB(4:6)';
 48    uGoalP = TB2P(1:3,1:3)*nGoalCartesianB(1:3)' + TB2P(1:3,4);
 49    uDotGoalP = TB2P(1:3,1:3)*nGoalCartesianB(4:6)';
 50    
 51    %Transform the Cartesian goal and final positions in the pan coordinate
 52    %frame to the joint space.
 53    qInit = sherpaTTIK(uInitP,kC,jointLimits);
 54    qGoal = sherpaTTIK(uGoalP,kC,jointLimits);
 55    qDotInit = sherpaTTIKVel(uDotInitP',qInit',kC);
 56    qDotGoal = sherpaTTIKVel(uDotGoalP',qGoal',kC);
 57    
 58    nInitJoint = [qInit phiInit 0 qDotInit' 0 omegaInit];
 59    nGoalJoint = [qGoal 0 0 qDotGoal' 0 0];
 60    
 61    %Check that the initial and final positions are valid. If they are not
 62    %return failure and an empty path.
 63    if (validJointState(nInitJoint,jointLimits) && validJointState(nGoalJoint,jointLimits))
 64        success = true;
 65        %Run buildRRT.
 66        nInit = [1 0 0 nInitJoint];
 67        nGoal = [0 0 0 nGoalJoint];
 68        [T,pathJ] = buildRRT(nInit,nGoal,NUM_NODES,jointLimits,cartesianLimits,panHeight,HGAINS,NODE_SIZE,U,U_SIZE,dt,Dt,kC,ankleThreshold,exhaustive,threshold,goalSeedFreq,uBDot,legNum);
 69        %Transform path back to the Cartesian space.
 70        pathC = transformPath(pathJ,kC,TP2B);
 71    else
 72        success = false;
 73        pathC = [];
 74        pathJ = [];
 75        T = [];
 76    end
 77    
 78    %Linearly interpolate to the goal state from the final state.
 79    if isempty(pathC)
 80        sFinalC = [0 0 nInitCartesianB true];
 81    else
 82        sFinalC = pathC(end,:);
 83    end
 84    sGoalC = [0 0 nGoalCartesianB true];
 85    pathCorrection = linInterp(sFinalC,sGoalC,10);
 86    pathC = [pathC; pathCorrection];
 87    [h,~] = size(pathCorrection);
 88    for i = 1:h
 89        uB = TB2P(1:3,1:3)*pathCorrection(i,3:5)' + TB2P(1:3,4);
 90        q = sherpaTTIK(uB',kC,jointLimits);
 91        pathJ = [pathJ; [pathCorrection(i,1) q 0 0 0 0 0 0 0]];  
 92    end
 93
 94end

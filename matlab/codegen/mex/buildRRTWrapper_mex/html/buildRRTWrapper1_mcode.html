1%BUILDRRTWRAPPER This function acts as a wrapper for the buildRRT function.
  2%Code generation for the singleLegPlanner is performed using this function
  3%as an entry point.
  4%  
  5%Inputs:
  6%-nInitCartesianB: the
  7%-nGoalCartesianB:
  8%-jointLimits:
  9%-bodyHeight:
 10%-U:
 11%-dt:
 12%-Dt:
 13%-kC:
 14%-threshold:
 15%-legNum:
 16%-uBDot:
 17%
 18%Outputs:
 19%-T:
 20%-pathC:
 21%-pathJ:
 22%-success:
 23%
 24%buildRRTWrapper.m
 25%author: wreid
 26%date: 20150502
 27
 28function [T,pathC,pathJ,success] = buildRRTWrapper(nInitCartesianB,nGoalCartesianB,phiInit,omegaInit,jointLimits,bodyHeight,U,dt,Dt,kC,legNum,uBDot)
 29
 30    persistent NUM_NODES 
 31    persistent NODE_SIZE 
 32    persistent U_SIZE 
 33    persistent ankleThreshold 
 34    persistent exhaustive 
 35    persistent goalSeedFreq 
 36    persistent cartesianLimits
 37    persistent HGAINS
 38    persistent threshold
 39    
 40    if isempty(NUM_NODES)
 41        NUM_NODES = int32(2000);
 42    end
 43    if isempty(NODE_SIZE)
 44        NODE_SIZE = int32(13);
 45    end
 46    if isempty(U_SIZE)
 47        U_SIZE = int32(9);
 48    end
 49    if isempty(ankleThreshold)
 50        ankleThreshold = 5*pi/180;
 51    end
 52    if isempty(exhaustive)
 53        exhaustive = false;
 54    end
 55    if isempty(goalSeedFreq)
 56        goalSeedFreq = int32(20);
 57    end
 58    if isempty(cartesianLimits)
 59        cartesianLimits = [-0.2930   -1.1326   -0.6710   -0.7546];
 60    end
 61    if isempty(HGAINS)
 62        HGAINS = [1 0 0.5];
 63    end
 64    if isempty(threshold)
 65        threshold = 0.005;
 66    end
 67    
 68    panHeight  = getPanHeight(bodyHeight,kC);
 69
 70    %Transform the nInitCartesianB and nGoalCartesianB variables from the body coordinate frame
 71    %to the pan coordinate frame.
 72    TP2B = trP2B(kC,legNum);
 73    TB2P = trInv(TP2B); %inv(TP2B);%
 74    uInitP = TB2P(1:3,1:3)*nInitCartesianB(1:3)' + TB2P(1:3,4);
 75    uDotInitP = TB2P(1:3,1:3)*nInitCartesianB(4:6)';
 76    uGoalP = TB2P(1:3,1:3)*nGoalCartesianB(1:3)' + TB2P(1:3,4);
 77    uDotGoalP = TB2P(1:3,1:3)*nGoalCartesianB(4:6)';
 78    
 79    %Transform the Cartesian goal and final positions in the pan coordinate
 80    %frame to the joint space.
 81    qInit = sherpaTTIK(uInitP,kC,jointLimits);
 82    qGoal = sherpaTTIK(uGoalP,kC,jointLimits);
 83    qDotInit = sherpaTTIKVel(uDotInitP',qInit',kC);
 84    qDotGoal = sherpaTTIKVel(uDotGoalP',qGoal',kC);
 85    
 86    nInitJoint = [qInit phiInit 0 qDotInit' 0 omegaInit];
 87    nGoalJoint = [qGoal 0 0 qDotGoal' 0 0];
 88    
 89    %Check that the initial and final positions are valid. If they are not
 90    %return failure and an empty path.
 91    if (validJointState(nInitJoint,jointLimits) && validJointState(nGoalJoint,jointLimits))
 92        success = true;
 93        %Run buildRRT.
 94        nInit = [1 0 0 nInitJoint];
 95        nGoal = [0 0 0 nGoalJoint];
 96        [T,pathJ] = buildRRT(nInit,nGoal,NUM_NODES,jointLimits,cartesianLimits,panHeight,HGAINS,NODE_SIZE,U,U_SIZE,dt,Dt,kC,ankleThreshold,exhaustive,threshold,goalSeedFreq,uBDot,legNum);
 97        %Transform path back to the Cartesian space.
 98        pathC = transformPath(pathJ,kC,TP2B);
 99    else
100        success = false;
101        pathC = [];
102        pathJ = [];
103        T = [];
104    end
105    
106    %Linearly interpolate to the goal state from the final state.
107    sFinalC = pathC(end,:);
108    sGoalC = [0 0 nGoalCartesianB true];
109    pathCorrection = linInterp(sFinalC,sGoalC,10);
110    pathC = [pathC; pathCorrection];
111    [h,~] = size(pathCorrection);
112    for i = 1:h
113        uB = TB2P(1:3,1:3)*pathCorrection(i,3:5)' + TB2P(1:3,4);
114        q = sherpaTTIK(uB',kC,jointLimits);
115        pathJ = [pathJ; [pathCorrection(i,1) q 0 0 0 0 0 0 0]];  
116    end
117
118end

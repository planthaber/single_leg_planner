119function pathC = transformPath(pathJ,kC,TP2B)
120    [pathH,~] = size(pathJ);
121    pathC = zeros(pathH,9);
122    dist2Go = 0;
123    for i = 1:pathH
124        uP = sherpaTTFK(pathJ(i,2:4),kC);
125        uPDot = sherpaTTFKVel(pathJ(i,7:9),pathJ(i,2:4),kC);
126        uB = TP2B(1:3,1:3)*uP' + TP2B(1:3,4);
127        uBDot = TP2B(1:3,1:3)*uPDot;
128        if i ~= 1
129            dist2Go = dist2Go + norm(uB'-pathC(i-1,3:5));
130        end
131        pathC(i,:) = [pathJ(i,1) dist2Go uB' uBDot' false];
132    end
133end

1function s = getPhiAndOmega(uBDot,s,kC,legNum)
 2
 3    uG = [0 0 0];
 4
 5    %Homogeneous transformation matrices.
 6    [~,TP2B,TI2P,TJ2I,TO2J,TQ2O,TR2Q,TS2R,~,~] = generateTrMatrices(uG,[s.alpha s.beta s.gamma 0],kC,legNum);
 7
 8    TS2B = TP2B*TI2P*TJ2I*TO2J*TQ2O*TR2Q*TS2R;
 9    TS2P = TI2P*TJ2I*TO2J*TQ2O*TR2Q*TS2R;
10    TS2I = TJ2I*TO2J*TQ2O*TR2Q*TS2R;
11    TS2O = TQ2O*TR2Q*TS2R;
12
13    TB2S = trInv(TS2B);
14    TP2S = trInv(TS2P);
15    TI2S = trInv(TS2I);
16    TO2S = trInv(TS2O);
17
18    %Adjunct transformation matrices.
19    AdB2S = tr2Adj(TB2S);
20    AdP2S = tr2Adj(TP2S);
21    AdI2S = tr2Adj(TI2S);
22    AdO2S = tr2Adj(TO2S);   
23
24    %Pan joint rate
25    vP = zeros(1,3);            %[m/s]
26    omegaP = [0 0  s.alphaDot];   %[rad/s]
27    uPDot = [vP omegaP]';
28
29    %Beta joint rate
30    vI = zeros(1,3);            %[m/s]
31    omegaI = [0 0  s.betaDot];    %[rad/s]
32    uIDot = [vI omegaI]';
33
34    %Gamma joint rate
35    vO = zeros(1,3);            %[m/s]
36    omegaO = [0 0  s.gammaDot];   %[rad/s]
37    uODot = [vO omegaO]';
38
39    %Velocity vector for the ankle frame.
40    uSDot = AdB2S*uBDot + AdP2S*uPDot + AdI2S*uIDot + AdO2S*uODot;
41    vS = uSDot(1:3);            %[m/s]
42
43    %Calculate the required phi joint angle and the required wheel speed,
44    %omega.
45    phi = atan2(vS(2),vS(1))-pi/2;   %[rad]
46    phi = mod(phi+pi,2*pi)-pi;
47    sign = 1;
48    if phi > pi/2
49        phi = phi - pi;
50        sign = -1;
51    elseif phi <= -pi/2
52        phi = phi + pi;
53        sign = -1;
54    end
55    omega = sign*norm(vS)/kC.r;      %[rad/s]
56    
57    s.phi = phi;
58    s.omega = omega;
59
60end

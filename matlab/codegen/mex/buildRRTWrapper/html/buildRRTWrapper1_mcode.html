1%BUILDRRTWRAPPER This function acts as a wrapper for the buildRRT function.
  2%Code generation for the singleLegPlanner is performed using this function
  3%as an entry point.
  4%  
  5%Inputs:
  6%-nInitCartesianB: the
  7%-nGoalCartesianB:
  8%-jointLimits:
  9%-bodyHeight:
 10%-U:
 11%-dt:
 12%-Dt:
 13%-kC:
 14%-threshold:
 15%-legNum:
 16%
 17%Outputs:
 18%-T:
 19%-pathC:
 20%-pathJ:
 21%-success:
 22%
 23%buildRRTWrapper.m
 24%author: wreid
 25%date: 20150502
 26
 27function [T,pathC,pathJ,success] = buildRRTWrapper(nInitCartesianB,nGoalCartesianB,phiInit,omegaInit,jointLimits,bodyHeight,U,dt,Dt,kC,threshold,legNum,uBDot,HGAINS)
 28
 29    persistent NUM_NODES 
 30    persistent NODE_SIZE 
 31    persistent U_SIZE 
 32    persistent ankleThreshold 
 33    persistent exhaustive 
 34    persistent goalSeedFreq 
 35    persistent cartesianLimits
 36    
 37    if isempty(NUM_NODES)
 38        NUM_NODES = int32(1000);
 39    end
 40    if isempty(NODE_SIZE)
 41        NODE_SIZE = int32(11);
 42    end
 43    if isempty(U_SIZE)
 44        U_SIZE = int32(5);
 45    end
 46    if isempty(ankleThreshold)
 47        ankleThreshold = pi/8;
 48    end
 49    if isempty(exhaustive)
 50        exhaustive = false;
 51    end
 52    if isempty(goalSeedFreq)
 53        goalSeedFreq = int32(20);
 54    end
 55    
 56    if isempty(cartesianLimits)
 57        cartesianLimits = zeros(1,4);
 58        betaMin = jointLimits(1,2);
 59        betaMax = jointLimits(2,2);
 60        gammaMin = jointLimits(1,3);
 61        gammaMax = jointLimits(2,3);
 62        u1 = sherpaTTFK([0 betaMin gammaMin],kC);
 63        u2 = sherpaTTFK([0 betaMax gammaMax],kC);
 64        u3 = sherpaTTFK([0 betaMin gammaMax],kC);
 65        u4 = sherpaTTFK([0 betaMax gammaMin],kC);
 66        cartesianLimits(1) = u1(3);
 67        cartesianLimits(2) = u2(3);
 68        cartesianLimits(3) = u3(3);
 69        cartesianLimits(4) = u4(3);
 70    end
 71
 72    
 73    panHeight  = getPanHeight(bodyHeight,kC);
 74
 75    %Transform the nInitCartesianB and nGoalCartesianB variables from the body coordinate frame
 76    %to the pan coordinate frame.
 77    TP2B = trP2B(kC,legNum);
 78    TB2P = trInv(TP2B); %inv(TP2B);%
 79    uInitP = TB2P(1:3,1:3)*nInitCartesianB(1:3)' + TB2P(1:3,4);
 80    uDotInitP = TB2P(1:3,1:3)*nInitCartesianB(4:6)';
 81    uGoalP = TB2P(1:3,1:3)*nGoalCartesianB(1:3)' + TB2P(1:3,4);
 82    uDotGoalP = TB2P(1:3,1:3)*nGoalCartesianB(4:6)';
 83    
 84    %Transform the Cartesian goal and final positions in the pan coordinate
 85    %frame to the joint space.
 86    qInit = sherpaTTIK(uInitP,kC,jointLimits);
 87    qGoal = sherpaTTIK(uGoalP,kC,jointLimits);
 88    qDotInit = sherpaTTIKVel(uDotInitP',qInit',kC);
 89    qDotGoal = sherpaTTIKVel(uDotGoalP',qGoal',kC);
 90    
 91    nInitJoint = [qInit qDotInit'];
 92    nGoalJoint = [qGoal qDotGoal'];
 93    
 94    %Check that the initial and final positions are valid. If they are not
 95    %return failure and an empty path.
 96    if (validJointState(nInitJoint,jointLimits) && validJointState(nGoalJoint,jointLimits))
 97        success = true;
 98        %Run buildRRT.
 99        nInit = [0 0 0 nInitJoint 0 0];
100        nGoal = [0 0 0 nGoalJoint 0 0];
101        [T,pathJ] = buildRRT(nInit,nGoal,NUM_NODES,jointLimits,cartesianLimits,panHeight,HGAINS,NODE_SIZE,U,U_SIZE,dt,Dt,kC,ankleThreshold,exhaustive,threshold,goalSeedFreq,uBDot,legNum);
102        %Transform path back to the Cartesian space.
103        [pathC,pathJ] = transformPath(pathJ,NODE_SIZE,kC,dt,Dt,TP2B);
104        pathC = [0 nInitCartesianB; pathC];
105        pathJ = [0 qInit qDotInit' phiInit omegaInit; pathJ];
106    else
107        success = false;
108        pathC = [];
109        pathJ = [];
110        T = [];
111    end 
112
113end

10function s = makeState(id,parentID,cost,alpha,beta,gamma,alphaDot,betaDot,gammaDot,phi,omega,dt,Dt)
11
12    numIterations = round(Dt/dt);
13    transitionArray = makeEmptyTransitionState();
14    for i = 1:numIterations-1
15        transitionArray = [transitionArray makeEmptyTransitionState()];
16    end
17
18    s = struct('id',id,...
19                'parentID',parentID,...
20                'cost',cost,...
21                'alpha',alpha,...
22                'beta',beta,...
23                'gamma',gamma,...
24                'alphaDot',alphaDot,...
25                'betaDot',betaDot,...
26                'gammaDot',gammaDot,...
27                'phi',phi,...
28                'omega',omega,...
29                'transitionArray',transitionArray);
30end

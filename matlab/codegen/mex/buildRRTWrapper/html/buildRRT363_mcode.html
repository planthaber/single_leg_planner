 5function [T,path] = buildRRT(nInit,nGoal,NUM_NODES,jointLimits,cartesianLimits,panHeight,HGAINS,NODE_SIZE,U,U_SIZE,dt,Dt,kC,ankleThreshold,exhaustive,threshold,goalSeedFreq,uBDot,legNum)
 6%buildRRT Icrementally builds a rapidly exploring random tree.
 7%   An RRT is build by incrementally selecting a random state from the
 8%   available state space as defined by the MIN and MAX vectors. The tree is
 9%   started at xInit and is extended until the number of maximum nodes, K has
10%   been reached. A path is selected if the goal region as defined by xGoal
11%   has been reached by the RRT.
12
13    %%TODO: Make a structure input for nInit and nGoal
14    %%TEMPORARY definition of init and goal nodes.
15    %nInit = makeNode(1,0,0,nInit(4:6),nInit(7:9),)
16
17    %Constant Declaration                                                      
18    transitionArrayLength = (round(Dt/dt)+1)*6;    
19    
20    %Variable Initialization
21    T = zeros(NUM_NODES,NODE_SIZE+transitionArrayLength);     %Define a zero array that will be used to 
22                                            %store data from each tree node.
23    T(1,:) = [nInit zeros(1,transitionArrayLength)];                         %Initialize the tree with initial state.
24    nodeIDCount = 1;
25    jointRange = jointLimits(2,:) - jointLimits(1,:);
26    
27    if ~exhaustive
28    
29        for i = 2:NUM_NODES
30            [T,nodeIDCount] = rrtLoop(T,jointLimits,cartesianLimits,kC,panHeight,U,Dt,dt,NODE_SIZE,U_SIZE,HGAINS,ankleThreshold,nodeIDCount,nGoal,goalSeedFreq,uBDot,legNum);
31        end
32        
33    else
34        %TODO: calculate maximum distance given the configuration space.
35        dist = 100;
36        %TODO: make the threshold distance
37        while dist > threshold && nodeIDCount < NUM_NODES 
38            [T,nodeIDCount] = rrtLoop(T,jointLimits,cartesianLimits,kC,panHeight,U,Dt,dt,NODE_SIZE,U_SIZE,HGAINS,ankleThreshold,nodeIDCount,nGoal,goalSeedFreq,uBDot,legNum);
39            [~,~,dist] = nearestNeighbour(nGoal,T,HGAINS,jointLimits,kC,nodeIDCount,NODE_SIZE);
40            if mod(nodeIDCount,100) == 0
41                %fprintf('PROGRESS STATUS: dist = %.3f\n',dist);
42            end
43        end   
44    end
45    
46    %Find the closest node in the tree to the goal node.
47    [xNearest,transitionArrayNearest,~] = nearestNeighbour(nGoal,T,HGAINS,jointLimits,kC,nodeIDCount,NODE_SIZE);
48        
49    check = xNearest(1);
50    next = xNearest;
51    path = [next transitionArrayNearest];
52    while check ~= 0 && next(2) ~= 0
53        next = T(next(2),:);
54        path = [path; next];
55        check = next(2);
56    end
57    
58end

 5function d = heuristicSingleLeg(xA,xB,HGAINS,jointLimits,kC)
 6%heuristic Calculates the distance between states x1 and x2.
 7
 8    alphaA = xA(4); 
 9    betaA = xA(5);
10    gammaA = xA(6);
11    
12    alphaB = xB(4);
13    betaB = xB(5);
14    gammaB = xB(6);
15    
16    alphaDotA = xA(7);
17    betaDotA = xA(8);
18    gammaDotA = xA(9);
19    
20    alphaDotB = xB(7);
21    betaDotB = xB(8);
22    gammaDotB = xB(9);
23    
24    %Calculate the distance between angular positions.
25    xStarMin = legRadius(jointLimits(1,2),jointLimits(1,3),kC);
26    xStarMax = legRadius(jointLimits(2,2),jointLimits(2,3),kC);
27    
28    dxStarMax = xStarMax-xStarMin;
29    dAlphaMax = angDiff(jointLimits(1,1),jointLimits(1,2));
30    
31    dPosMax = posMetric(xStarMin,dxStarMax,dAlphaMax);
32    
33    xStarA = legRadius(betaA,gammaA,kC);
34    xStarB = legRadius(betaB,gammaB,kC);
35    
36    dxStar = xStarB-xStarA;
37    dAlpha = angDiff(alphaA,alphaB);
38    
39    dPos = sqrt(dxStar^2+xStarA^2*dAlpha^2);
40    dPosNorm = HGAINS(1)*dPos/dPosMax;
41    
42    %Calculate the total distance.
43    d = dPos; %dPosNorm+dVelNorm 
44    
45end

 5function [T,path] = buildRRT(nInit,nGoal,NUM_NODES,jointLimits,cartesianLimits,panHeight,HGAINS,NODE_SIZE,U,U_SIZE,dt,Dt,kC,ankleThreshold,exhaustive,threshold,goalSeedFreq,uBDot,legNum)
 6%buildRRT Icrementally builds a rapidly exploring random tree.
 7%   An RRT is build by incrementally selecting a random state from the
 8%   available state space as defined by the MIN and MAX vectors. The tree is
 9%   started at xInit and is extended until the number of maximum nodes, K has
10%   been reached. A path is selected if the goal region as defined by xGoal
11%   has been reached by the RRT.
12
13    %Constant Declaration                                                      
14    transitionArrayLength = (round(Dt/dt)+1)*10;    
15    
16    %Variable Initialization
17    T = zeros(NUM_NODES,NODE_SIZE+transitionArrayLength);     %Define a zero array that will be used to 
18                                            %store data from each tree node.
19    T(1,:) = [nInit zeros(1,transitionArrayLength)];                         %Initialize the tree with initial state.
20    nodeIDCount = 1;
21    
22    
23    if ~exhaustive
24    
25        for i = 2:NUM_NODES
26            [T,nodeIDCount] = rrtLoop(T,jointLimits,cartesianLimits,kC,panHeight,U,Dt,dt,NODE_SIZE,U_SIZE,HGAINS,ankleThreshold,nodeIDCount,nGoal,goalSeedFreq,uBDot,legNum);
27        end
28        
29    else
30        %TODO: calculate maximum distance given the configuration space.
31        dist = 100; %[m]
32        %TODO: make the threshold distance
33        while dist > threshold && nodeIDCount < NUM_NODES 
34            [T,nodeIDCount] = rrtLoop(T,jointLimits,cartesianLimits,kC,panHeight,U,Dt,dt,NODE_SIZE,U_SIZE,HGAINS,ankleThreshold,nodeIDCount,nGoal,goalSeedFreq,uBDot,legNum);
35            [~,~,dist] = nearestNeighbour(nGoal,T,HGAINS,jointLimits,kC,nodeIDCount,NODE_SIZE);
36            if mod(nodeIDCount,100) == 0
37                %fprintf('PROGRESS STATUS: dist = %.3f\n',dist);
38            end
39        end   
40    end
41    
42    %Find the closest node in the tree to the goal node.
43    [xNearest,transitionArrayNearest,~] = nearestNeighbour(nGoal,T,HGAINS,jointLimits,kC,nodeIDCount,NODE_SIZE);
44        
45    check = xNearest(1);
46    next = xNearest;
47    path = [];
48    transitionArray = transitionArrayNearest;
49    
50    %Iterate over the tree until the initial state has been found.
51    while check ~= 0 && next(2) ~=0
52
53        transitionPath = [];
54        for i = 1:10:length(transitionArray)
55            transitionPath = [transitionPath; transitionArray(i:i+9)];
56        end
57        path = [transitionPath; path];
58        
59        next = T(next(2),:);
60        check = next(2);
61        
62        transitionArray = next(14:end);
63    end
64    
65    [pathH,~] = size(path);
66    t = dt*(1:pathH)';
67    path = [t path];
68    
69end

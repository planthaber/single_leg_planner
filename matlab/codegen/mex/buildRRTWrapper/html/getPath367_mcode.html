68function path = getPath(xNearest,T,dt)
69    check = xNearest.id;
70    next = xNearest;
71    path = [];
72    timeCount = 0;
73    while check ~= 0
74        next = T(next.parentID);
75        transition = [];
76        for i = 1:length(next.transitionArray)
77            s = next.transitionArray(i);
78            transition = [0 s.alpha s.beta s.gamma s.alphaDot s.betaDot s.gammaDot s.phi s.omega; transition];
79        end
80        path = [path; transition];
81        check = next.parentID;
82    end
83    
84    timeCount = 0;
85    [h,~] = size(path);
86    for i = 1:h 
87        path(i,1) = timeCount;
88        timeCount = timeCount + dt;
89    end
90end

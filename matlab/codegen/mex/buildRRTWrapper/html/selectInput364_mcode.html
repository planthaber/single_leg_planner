1function sNew = selectInput(sNear,sRand,U,dt,Dt,U_SIZE,HGAINS,kC,ankleThreshold,jointLimits,uBDot,legNum)
 2%selectInput Selects the most appropriate control input.
 3%   A control input is selected from a set of control inputs, U. An input
 4%   is selected by applying each of the inputs to to state xNear, which
 5%   results in p candidate states, where p is the size of the input set.
 6%   The control input corresponding to candidate state that is closest to
 7%   x1 is returned as u.
 8
 9    %Initialize arrays to store the candidate new state data and the
10    %distances between each candidate state and the xNear state.
11    %candStates = zeros(U_SIZE,NODE_SIZE);
12    %candTransArrays = zeros(U_SIZE,(round(Dt/dt)+1)*6);
13    
14    distance = zeros(1,U_SIZE);
15    
16    phiNear = sNear.phi;
17    
18    UJoint = zeros(U_SIZE,3);
19
20    candStates = makeEmptyState(dt,Dt);
21    for i = 1:U_SIZE-1
22        candStates = [candStates makeEmptyState(dt,Dt)];
23    end
24    %Transform the control inputs to joint space.
25    for i = 1:U_SIZE
26        gammaDotDot = getConstrainedGammaDotDot(kC,U(i,:),sNear);
27        UJoint(i,:) = [U(i,:) gammaDotDot];
28    end    
29
30    %Increment over the control vector. Generate a candidate state for each
31    %possible control input.
32    for i = 1:U_SIZE
33        
34        %Generate a candidate state using a fourth order Runge-Kutta 
35        %integration technique.
36        candStates(i) = rk4(UJoint(i,:),dt,Dt,sNear,jointLimits,uBDot,kC,legNum);
37       
38        %Calculate the distance between the candidate state and the random
39        %state.
40        hDiff = heuristicSingleLeg(candStates(i),sRand,HGAINS,jointLimits,kC);
41
42        %Apply the ankle constraint to penalize any candidate state that
43        %requires a change of ankle position greater than the allowed ankle
44        %movement in a single time step.
45        aGain = HGAINS(3);
46        
47        candStates(i) = getPhiAndOmega(uBDot,candStates(i),kC,legNum);
48        
49        aDiff = angDiff(phiNear,candStates(i).phi);
50        ankleDiffMax = pi;
51        
52        if aDiff > ankleThreshold
53            aDiff = 1;
54        else
55            aDiff = abs(aDiff/ankleDiffMax);
56        end
57        
58        %Calculate a distance metric that includes the heurisitc distance
59        %as well as any penalty due to ankle movements.
60        distance(i) = (1-aGain)*hDiff + aGain*aDiff;
61        %distance(i) = hDiff;
62    end
63    
64    
65    [~,minIndex] = min(distance);
66    sNew = candStates(minIndex);
67    
68end

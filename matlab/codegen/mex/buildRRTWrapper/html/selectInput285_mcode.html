1function [xNew,transitionArray] = selectInput(xNear,xRand,U,dt,Dt,NODE_SIZE,U_SIZE,HGAINS,kC,ankleThreshold,jointLimits)
 2%selectInput Selects the most appropriate control input.
 3%   A control input is selected from a set of control inputs, U. An input
 4%   is selected by applying each of the inputs to to state xNear, which
 5%   results in p candidate states, where p is the size of the input set.
 6%   The control input corresponding to candidate state that is closest to
 7%   x1 is returned as u.
 8
 9    %Initialize arrays to store the candidate new state data and the
10    %distances between each candidate state and the xNear state.
11    candStates = zeros(U_SIZE,NODE_SIZE);
12    candTransArrays = zeros(U_SIZE,(round(Dt/dt)+1)*6);
13    distance = zeros(1,U_SIZE);
14    
15    qANear = xNear(end);
16    
17    UJoint = zeros(U_SIZE,3);
18    
19    %Transform the control inputs to joint space.
20    for i = 1:U_SIZE
21        %gammaDotDot = (-betaDotDot*L3*cos(beta)+betaDot^2*L3*sin(beta)+gammaDot^2*L5*sin(zeta+gamma))/(L5*cos(zeta+gamma));
22        gammaDotDot = getConstrainedGammaDotDot(kC,U(i,:),xNear(7:9),xNear(4:6));
23        UJoint(i,:) = [U(i,:) gammaDotDot];
24    end    
25
26    %Increment over the control vector. Generate a candidate state for each
27    %possible control input.
28    for i = 1:U_SIZE
29        
30        %Generate a candidate state using a fourth order Runge-Kutta 
31        %integration technique.
32        [candStates(i,:),candTransArrays(i,:)] = rk4(UJoint(i,:),dt,Dt,xNear,jointLimits);
33        %U_check = sherpaTTFKAcc(U_joint(i,:)',xNear(7:9)',xNear(4:6)',kinematicConst)
34        %velCheck = sherpaTTFKVel(candStates(i,7:9)',candStates(i,4:6)',kinematicConst);
35       
36        %Calculate the distance between the candidate state and the random
37        %state.
38        hDiff = heuristicSingleLeg(candStates(i,:),xRand,HGAINS,jointLimits,kC);
39
40        %Apply the ankle constraint to penalize any candidate state that
41        %requires a change of ankle position greater than the allowed ankle
42        %movement in a single time step.
43        aGain = HGAINS(3);
44        if xNear(1) == 1
45            aGain = 0;
46        end
47        
48        %Ss = [1000 1];
49        %qWDot = 1;
50        %r = 0.378/2;
51        
52        [candStates(i,10),candStates(i,11)] = calcPhi(candStates(i,7:9),candStates(i,4:6),kC);%,Ss,qWDot,r);
53        aDiff = angDiff(qANear,candStates(i,10));
54        ankleDiffMax = pi;
55        
56        if aDiff > ankleThreshold
57            aDiff = 1;
58        else
59            aDiff = abs(aDiff/ankleDiffMax);
60            aDiff = 0;
61        end
62        
63        %Calculate a distance metric that includes the heurisitc distance
64        %as well as any penalty due to ankle movements.
65        distance(i) = (1-aGain)*hDiff + aGain*aDiff;
66        %distance(i) = hDiff;
67    end
68    
69    
70    [~,minIndex] = min(distance);
71    xNew = candStates(minIndex,:);
72    transitionArray = candTransArrays(minIndex,:);
73    
74    %velCheck = sherpaTTFKVel(xNew(7:9)',xNew(4:6)',kinematicConst)
75
76end

 5function d = heuristicSingleLeg(xA,xB,HGAINS,jointLimits,kinematicConst)
 6%heuristic Calculates the distance between states x1 and x2.
 7
 8    L1 = kinematicConst(1);
 9    L2 = kinematicConst(2);
10    L3 = kinematicConst(3);
11    L4 = kinematicConst(4);
12    L5 = kinematicConst(5);
13    L6 = kinematicConst(6);
14    L7 = kinematicConst(7);
15    L8 = kinematicConst(8);
16    zeta = kinematicConst(9);
17
18    alphaA = xA(4); 
19    betaA = xA(5);
20    gammaA = xA(6);
21    
22    alphaB = xB(4);
23    betaB = xB(5);
24    gammaB = xB(6);
25    
26    alphaDotA = xA(7);
27    betaDotA = xA(8);
28    gammaDotA = xA(9);
29    
30    alphaDotB = xB(7);
31    betaDotB = xB(8);
32    gammaDotB = xB(9);
33    
34    %Calculate the distance between angular positions.
35    xStarMin = legRadius(jointLimits(1,2),jointLimits(1,3),zeta,L2,L3,L4,L5,L7);
36    xStarMax = legRadius(jointLimits(2,2),jointLimits(2,3),zeta,L2,L3,L4,L5,L7);
37    
38    dxStarMax = xStarMax-xStarMin;
39    dAlphaMax = angDiff(jointLimits(1,1),jointLimits(1,2));
40    
41    dPosMax = posMetric(xStarMin,dxStarMax,dAlphaMax);
42    
43    xStarA = legRadius(betaA,gammaA,zeta,L2,L3,L4,L5,L7);
44    xStarB = legRadius(betaB,gammaB,zeta,L2,L3,L4,L5,L7);
45    
46    dxStar = xStarB-xStarA;
47    dAlpha = angDiff(alphaA,alphaB);
48    
49    dPos = sqrt(dxStar^2+xStarA^2*dAlpha^2);
50    dPosNorm = HGAINS(1)*dPos/dPosMax;
51    
52    %Calculate the total distance.
53    d = dPos; %dPosNorm+dVelNorm 
54    
55end

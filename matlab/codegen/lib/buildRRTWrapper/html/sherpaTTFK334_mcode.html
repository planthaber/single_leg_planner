 5function [x,y,z] = sherpaTTFK(alpha,beta,gamma,kinematicConst)
 6%sherpaTTFK Sherpa_TT Forward Kinematics
 7%   Calculates the x,y,z position of the contact point given the alpha,
 8%   beta and gamma joint values.
 9
10    L1 = kinematicConst(1);
11    L2 = kinematicConst(2);
12    L3 = kinematicConst(3);
13    L4 = kinematicConst(4);
14    L5 = kinematicConst(5);
15    L6 = kinematicConst(6);
16    L7 = kinematicConst(7);
17    L8 = kinematicConst(8);
18    zeta = kinematicConst(9);
19    
20    x = (L2 + L3*cos(-beta)+L4*cos(zeta)+L5*cos(gamma+zeta)-L7).*cos(alpha);
21    y = (L2 + L3*cos(-beta)+L4*cos(zeta)+L5*cos(gamma+zeta)-L7).*sin(alpha);
22    z = (L1 + L3*sin(-beta)-L4*sin(zeta)-L5*sin(gamma+zeta)-L6-L8);
23   
24end

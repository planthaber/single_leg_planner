1function [xNew,transitionArray] = selectInput(xNear,xRand,U,dt,Dt,NODE_SIZE,U_SIZE,HGAINS,kinematicConst,ankleThreshold,jointLimits)
 2%selectInput Selects the most appropriate control input.
 3%   A control input is selected from a set of control inputs, U. An input
 4%   is selected by applying each of the inputs to to state xNear, which
 5%   results in p candidate states, where p is the size of the input set.
 6%   The control input corresponding to candidate state that is closest to
 7%   x1 is returned as u.
 8
 9    %Initialize arrays to store the candidate new state data and the
10    %distances between each candidate state and the xNear state.
11    candStates = zeros(U_SIZE,NODE_SIZE);
12    candTransArrays = zeros(U_SIZE,(round(Dt/dt)+1)*6);
13    distance = zeros(1,U_SIZE);
14    
15    qANear = xNear(end);
16    
17    U_joint = zeros(U_SIZE,3);
18    
19    %Transform the control inputs to joint space.
20    for i = 1:U_SIZE
21        betaDotDot = U(i,2);
22        betaDot = xNear(8);
23        gammaDot = xNear(9);
24        beta = xNear(5);
25        gamma = xNear(6);
26        L3 = kinematicConst(3);
27        L5 = kinematicConst(5);
28        zeta = kinematicConst(9);
29        gammaDotDot = (-betaDotDot*L3*cos(beta)+betaDot^2*L3*sin(beta)+gammaDot^2*L5*sin(zeta+gamma))/(L5*cos(zeta+gamma));
30        U_joint(i,:) = [U(i,:) gammaDotDot];
31
32        %U_check = sherpaTTFKAcc(U_joint(i,:)',xNear(7:9)',xNear(4:6)',kinematicConst)
33    end    
34
35    %Increment over the control vector. Generate a candidate state for each
36    %possible control input.
37    for i = 1:U_SIZE
38        
39        %Generate a candidate state using a fourth order Runge-Kutta 
40        %integration technique.
41        [candStates(i,:),candTransArrays(i,:)] = rk4(U_joint(i,:),dt,Dt,xNear,jointLimits);
42        %U_check = sherpaTTFKAcc(U_joint(i,:)',xNear(7:9)',xNear(4:6)',kinematicConst)
43        velCheck = sherpaTTFKVel(candStates(i,7:9)',candStates(i,4:6)',kinematicConst);
44       
45        %Calculate the distance between the candidate state and the random
46        %state.
47        hDiff = heuristicSingleLeg(candStates(i,:),xRand,HGAINS,jointLimits,kinematicConst);
48
49        %Apply the ankle constraint to penalize any candidate state that
50        %requires a change of ankle position greater than the allowed ankle
51        %movement in a single time step.
52        aGain = HGAINS(3);
53        if xNear(1) == 1
54            aGain = 0;
55        end
56        
57        %Ss = [1000 1];
58        %qWDot = 1;
59        %r = 0.378/2;
60        
61        [candStates(i,10),candStates(i,11)] = calcPhi(candStates(i,7:9),candStates(i,4:6),kinematicConst);%,Ss,qWDot,r);
62        aDiff = angDiff(qANear,candStates(i,10));
63        ankleDiffMax = pi;
64        
65        if aDiff > ankleThreshold
66            aDiff = 1;
67        else
68            aDiff = abs(aDiff/ankleDiffMax);
69            aDiff = 0;
70        end
71        
72        %Calculate a distance metric that includes the heurisitc distance
73        %as well as any penalty due to ankle movements.
74        distance(i) = (1-aGain)*hDiff + aGain*aDiff;
75        %distance(i) = hDiff;
76    end
77    
78    
79    [~,minIndex] = min(distance);
80    xNew = candStates(minIndex,:);
81    transitionArray = candTransArrays(minIndex,:);
82    
83    %velCheck = sherpaTTFKVel(xNew(7:9)',xNew(4:6)',kinematicConst)
84
85end

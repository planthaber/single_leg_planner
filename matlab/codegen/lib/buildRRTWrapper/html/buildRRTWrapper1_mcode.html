  6function [T,pathC,pathJ,success] = buildRRTWrapper(nInitCartesianB,nGoalCartesianB,jointLimits,bodyHeight,U,dt,Dt,kinematicConst,threshold,legNum)
  7
  8    persistent NUM_NODES 
  9    persistent HGAINS
 10    persistent NODE_SIZE 
 11    persistent U_SIZE 
 12    persistent ankleThreshold 
 13    persistent exhaustive 
 14    persistent goalSeedFreq 
 15    
 16    if isempty(NUM_NODES)
 17        NUM_NODES = int32(1000);
 18    end
 19    if isempty(HGAINS)
 20        HGAINS = [1 0 0.5];
 21    end
 22    if isempty(NODE_SIZE)
 23        NODE_SIZE = int32(11);
 24    end
 25    if isempty(U_SIZE)
 26        U_SIZE = int32(5);
 27    end
 28    if isempty(ankleThreshold)
 29        ankleThreshold = pi/8;
 30    end
 31    if isempty(exhaustive)
 32        exhaustive = false;
 33    end
 34    if isempty(goalSeedFreq)
 35        goalSeedFreq = int32(20);
 36    end
 37    
 38    panHeight  = getPanHeight(bodyHeight,kinematicConst);
 39
 40    %Transform the nInitCartesianB and nGoalCartesianB variables from the body coordinate frame
 41    %to the pan coordinate frame.
 42    TP2B = trP2B(kinematicConst,legNum);
 43    TB2P = invHomoMatrix(TP2B); %inv(TP2B);%
 44    uInitP = TB2P(1:3,1:3)*nInitCartesianB(1:3)' + TB2P(1:3,4);
 45    uDotInitP = TB2P(1:3,1:3)*nInitCartesianB(4:6)';
 46    uGoalP = TB2P(1:3,1:3)*nGoalCartesianB(1:3)' + TB2P(1:3,4);
 47    uDotGoalP = TB2P(1:3,1:3)*nGoalCartesianB(4:6)';
 48    
 49    %Transform the Cartesian goal and final positions in the pan coordinate
 50    %frame to the joint space.
 51    [alphaInit,betaInit,gammaInit] = sherpaTTIK(uInitP(1),uInitP(2),uInitP(3),kinematicConst,jointLimits);
 52    [alphaGoal,betaGoal,gammaGoal] = sherpaTTIK(uGoalP(1),uGoalP(2),uGoalP(3),kinematicConst,jointLimits);
 53    qDotInit = sherpaTTIKVel(uDotInitP',[alphaInit betaInit gammaInit]',kinematicConst);
 54    qDotGoal = sherpaTTIKVel(uDotGoalP',[alphaGoal betaGoal gammaGoal]',kinematicConst);
 55    alphaDotInit = qDotInit(1); betaDotInit = qDotInit(2); gammaDotInit = qDotInit(3);
 56    alphaDotGoal = qDotGoal(1); betaDotGoal = qDotGoal(2); gammaDotGoal = qDotGoal(3);
 57    
 58    nInitJoint = [alphaInit betaInit gammaInit alphaDotInit betaDotInit gammaDotInit];
 59    nGoalJoint = [alphaGoal betaGoal gammaGoal alphaDotGoal betaDotGoal gammaDotGoal];
 60    
 61    %Check that the initial and final positions are valid. If they are not
 62    %return failure and an empty path.
 63    if (validState(nInitJoint,jointLimits) && validState(nGoalJoint,jointLimits))
 64        success = true;
 65        %Run buildRRT.
 66        nInit = [0 0 0 nInitJoint 0 0];
 67        nGoal = [0 0 0 nGoalJoint 0 0];
 68        [T,pathJ] = buildRRT(nInit,nGoal,NUM_NODES,jointLimits,panHeight,HGAINS,NODE_SIZE,U,U_SIZE,dt,Dt,kinematicConst,ankleThreshold,exhaustive,threshold,goalSeedFreq);
 69        %Transform path back to the Cartesian space.
 70        [pathC,pathJ] = transformPath(pathJ,NODE_SIZE,kinematicConst,dt,Dt,TP2B);
 71        pathC = [0 nInitCartesianB; pathC];
 72        pathJ = [0 alphaInit betaInit gammaInit qDotInit'; pathJ];
 73    else
 74        success = false;
 75        pathC = [];
 76        pathJ = [];
 77        T = [];
 78    end 
 79
 80end

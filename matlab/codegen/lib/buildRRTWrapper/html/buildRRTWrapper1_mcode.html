  6function [T,pathC,pathJ,success] = buildRRTWrapper(nInitCartesianB,nGoalCartesianB,jointLimits,bodyHeight,U,dt,Dt,kC,threshold,legNum)
  7
  8    persistent NUM_NODES 
  9    persistent HGAINS
 10    persistent NODE_SIZE 
 11    persistent U_SIZE 
 12    persistent ankleThreshold 
 13    persistent exhaustive 
 14    persistent goalSeedFreq 
 15    
 16    if isempty(NUM_NODES)
 17        NUM_NODES = int32(1000);
 18    end
 19    if isempty(HGAINS)
 20        HGAINS = [1 0 0.5];
 21    end
 22    if isempty(NODE_SIZE)
 23        NODE_SIZE = int32(11);
 24    end
 25    if isempty(U_SIZE)
 26        U_SIZE = int32(5);
 27    end
 28    if isempty(ankleThreshold)
 29        ankleThreshold = pi/8;
 30    end
 31    if isempty(exhaustive)
 32        exhaustive = false;
 33    end
 34    if isempty(goalSeedFreq)
 35        goalSeedFreq = int32(20);
 36    end
 37    
 38    panHeight  = getPanHeight(bodyHeight,kC);
 39
 40    %Transform the nInitCartesianB and nGoalCartesianB variables from the body coordinate frame
 41    %to the pan coordinate frame.
 42    TP2B = trP2B(kC,legNum);
 43    TB2P = invHomoMatrix(TP2B); %inv(TP2B);%
 44    uInitP = TB2P(1:3,1:3)*nInitCartesianB(1:3)' + TB2P(1:3,4);
 45    uDotInitP = TB2P(1:3,1:3)*nInitCartesianB(4:6)';
 46    uGoalP = TB2P(1:3,1:3)*nGoalCartesianB(1:3)' + TB2P(1:3,4);
 47    uDotGoalP = TB2P(1:3,1:3)*nGoalCartesianB(4:6)';
 48    
 49    %Transform the Cartesian goal and final positions in the pan coordinate
 50    %frame to the joint space.
 51    qInit = sherpaTTIK(uInitP,kC,jointLimits);
 52    qGoal = sherpaTTIK(uGoalP,kC,jointLimits);
 53    qDotInit = sherpaTTIKVel(uDotInitP',qInit',kC);
 54    qDotGoal = sherpaTTIKVel(uDotGoalP',qGoal',kC);
 55    
 56    nInitJoint = [qInit qDotInit'];
 57    nGoalJoint = [qGoal qDotGoal'];
 58    
 59    %Check that the initial and final positions are valid. If they are not
 60    %return failure and an empty path.
 61    if (validState(nInitJoint,jointLimits) && validState(nGoalJoint,jointLimits))
 62        success = true;
 63        %Run buildRRT.
 64        nInit = [0 0 0 nInitJoint 0 0];
 65        nGoal = [0 0 0 nGoalJoint 0 0];
 66        [T,pathJ] = buildRRT(nInit,nGoal,NUM_NODES,jointLimits,panHeight,HGAINS,NODE_SIZE,U,U_SIZE,dt,Dt,kC,ankleThreshold,exhaustive,threshold,goalSeedFreq);
 67        %Transform path back to the Cartesian space.
 68        [pathC,pathJ] = transformPath(pathJ,NODE_SIZE,kC,dt,Dt,TP2B);
 69        pathC = [0 nInitCartesianB; pathC];
 70        pathJ = [0 qInit qDotInit'; pathJ];
 71    else
 72        success = false;
 73        pathC = [];
 74        pathJ = [];
 75        T = [];
 76    end 
 77
 78end

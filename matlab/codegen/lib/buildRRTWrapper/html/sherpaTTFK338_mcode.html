 5function [x,y,z] = sherpaTTFK(alpha,beta,gamma,kinematicConst)
 6%sherpaTTFK Sherpa_TT Forward Kinematics
 7%   Calculates the x,y,z position of the contact point given the alpha,
 8%   beta and gamma joint values.
 9    
10    [L1,L2,L3,L4,L5,L6,L7,L8,zeta,~,~,~,~,~,~,~] = extractKinematicConstants(kinematicConst);
11
12    x = (L2 + L3*cos(-beta)+L4*cos(zeta)+L5*cos(gamma+zeta)-L7).*cos(alpha);
13    y = (L2 + L3*cos(-beta)+L4*cos(zeta)+L5*cos(gamma+zeta)-L7).*sin(alpha);
14    z = (L1 + L3*sin(-beta)-L4*sin(zeta)-L5*sin(gamma+zeta)-L6-L8);
15   
16end

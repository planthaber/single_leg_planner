 5function [uDot,uDotNew] = sherpaTTFKVel(qDot,q,kinematicConst)
 6%sherpaTTFKVel Sherpa_TT single leg forward velocity kinematics.
 7
 8    L1 = kinematicConst(1);
 9    L2 = kinematicConst(2);
10    L3 = kinematicConst(3);
11    L4 = kinematicConst(4);
12    L5 = kinematicConst(5);
13    L6 = kinematicConst(6);
14    L7 = kinematicConst(7);
15    L8 = kinematicConst(8);
16    zeta = kinematicConst(9);
17    
18    alpha = q(1);
19    beta = q(2);
20    gamma = q(3);
21    phi = 0;
22    
23    alphaDot = qDot(1);
24    betaDot = qDot(2);
25    gammaDot = qDot(3);
26
27    xDot =      - alphaDot*sin(alpha)*(L2 - L7 + L5*cos(gamma + zeta) + L3*cos(beta) + L4*cos(zeta))...
28                - betaDot*L3*cos(alpha)*sin(beta)...
29                - gammaDot*L5*sin(gamma + zeta)*cos(alpha);
30    yDot =      alphaDot*cos(alpha)*(L2 - L7 + L5*cos(gamma + zeta) + L3*cos(beta) + L4*cos(zeta))...
31                -gammaDot*L5*sin(gamma + zeta)*sin(alpha)...
32                -betaDot*L3*sin(alpha)*sin(beta);
33    zDot =      -betaDot*L3*cos(beta)...
34                -L5*gammaDot*cos(zeta+gamma);
35
36    uDot = [xDot yDot zDot]';
37
38end

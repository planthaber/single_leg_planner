 5function [T,path] = buildRRT(nInit,nGoal,NUM_NODES,jointLimits,panHeight,HGAINS,NODE_SIZE,U,U_SIZE,dt,Dt,kinematicConst,ankleThreshold,exhaustive,threshold,goalSeedFreq)
 6%buildRRT Icrementally builds a rapidly exploring random tree.
 7%   An RRT is build by incrementally selecting a random state from the
 8%   available state space as defined by the MIN and MAX vectors. The tree is
 9%   started at xInit and is extended until the number of maximum nodes, K has
10%   been reached. A path is selected if the goal region as defined by xGoal
11%   has been reached by the RRT.
12
13    %Constant Declaration                                                      
14    
15    transitionArrayLength = (round(Dt/dt)+1)*6;    
16    
17    %Variable Initialization
18    T = zeros(NUM_NODES,NODE_SIZE+transitionArrayLength);     %Define a zero array that will be used to 
19                                            %store data from each tree node.
20    T(1,:) = [nInit zeros(1,transitionArrayLength)];                         %Initialize the tree with initial state.
21    nodeIDCount = 1;
22    jointRange = jointLimits(2,:) - jointLimits(1,:);
23    
24    if ~exhaustive
25    
26        for i = 2:NUM_NODES
27            [T,nodeIDCount] = rrtLoop(T,jointRange,jointLimits,kinematicConst,panHeight,U,Dt,dt,NODE_SIZE,U_SIZE,HGAINS,ankleThreshold,nodeIDCount,nGoal,goalSeedFreq);
28        end
29        
30    else
31        %TODO: calculate maximum distance given the configuration space.
32        dist = 100;
33        %TODO: make the threshold distance
34        while dist > threshold && nodeIDCount < NUM_NODES 
35            [T,nodeIDCount] = rrtLoop(T,jointRange,jointLimits,kinematicConst,panHeight,U,Dt,dt,NODE_SIZE,U_SIZE,HGAINS,ankleThreshold,nodeIDCount,nGoal,goalSeedFreq);
36            [~,~,dist] = nearestNeighbour(nGoal,T,HGAINS,jointLimits,kinematicConst,nodeIDCount,NODE_SIZE);
37            if mod(nodeIDCount,100) == 0
38                %fprintf('PROGRESS STATUS: dist = %.3f\n',dist);
39            end
40        end   
41    end
42    
43    %Find the closest node in the tree to the goal node.
44    [xNearest,transitionArrayNearest,~] = nearestNeighbour(nGoal,T,HGAINS,jointLimits,kinematicConst,nodeIDCount,NODE_SIZE);
45        
46    check = xNearest(1);
47    next = xNearest;
48    path = [next transitionArrayNearest];
49    while check ~= 0 && next(2) ~= 0
50        next = T(next(2),:);
51        path = [path; next];
52        check = next(2);
53    end
54    
55end

1%BUILDRRTWRAPPER This function acts as a wrapper for the buildRRT function.
  2%Code generation for the singleLegPlanner is performed using this function
  3%as an entry point.
  4%  
  5%Inputs:
  6%-nInitCartesianB: the
  7%-nGoalCartesianB:
  8%-jointLimits:
  9%-bodyHeight:
 10%-U:
 11%-dt:
 12%-Dt:
 13%-kC:
 14%-threshold:
 15%-legNum:
 16%
 17%Outputs:
 18%-T:
 19%-pathC:
 20%-pathJ:
 21%-success:
 22%
 23%buildRRTWrapper.m
 24%author: wreid
 25%date: 20150502
 26
 27function [T,pathC,pathJ,success] = buildRRTWrapper(nInitCartesianB,nGoalCartesianB,phiInit,omegaInit,jointLimits,bodyHeight,U,dt,Dt,kC,threshold,legNum,uBDot,HGAINS,NUM_NODES,exhaustive,ankleThreshold,cartesianLimits)
 28
 29    %persistent NUM_NODES 
 30    persistent NODE_SIZE 
 31    persistent U_SIZE 
 32    %persistent ankleThreshold 
 33    %persistent exhaustive 
 34    persistent goalSeedFreq 
 35    %persistent cartesianLimits
 36    
 37    %if isempty(NUM_NODES)
 38    %    NUM_NODES = int32(1000);
 39    %end
 40    if isempty(NODE_SIZE)
 41        NODE_SIZE = int32(13);
 42    end
 43    if isempty(U_SIZE)
 44        U_SIZE = int32(5);
 45    end
 46    %if isempty(ankleThreshold)
 47    %    ankleThreshold = pi/8;
 48    %end
 49    %if isempty(exhaustive)
 50    %    exhaustive = false;
 51    %end
 52    if isempty(goalSeedFreq)
 53        goalSeedFreq = int32(20);
 54    end
 55    
 56    panHeight  = getPanHeight(bodyHeight,kC);
 57
 58    %Transform the nInitCartesianB and nGoalCartesianB variables from the body coordinate frame
 59    %to the pan coordinate frame.
 60    TP2B = trP2B(kC,legNum);
 61    TB2P = trInv(TP2B); %inv(TP2B);%
 62    uInitP = TB2P(1:3,1:3)*nInitCartesianB(1:3)' + TB2P(1:3,4);
 63    uDotInitP = TB2P(1:3,1:3)*nInitCartesianB(4:6)';
 64    uGoalP = TB2P(1:3,1:3)*nGoalCartesianB(1:3)' + TB2P(1:3,4);
 65    uDotGoalP = TB2P(1:3,1:3)*nGoalCartesianB(4:6)';
 66    
 67    %Transform the Cartesian goal and final positions in the pan coordinate
 68    %frame to the joint space.
 69    qInit = sherpaTTIK(uInitP,kC,jointLimits);
 70    qGoal = sherpaTTIK(uGoalP,kC,jointLimits);
 71    qDotInit = sherpaTTIKVel(uDotInitP',qInit',kC);
 72    qDotGoal = sherpaTTIKVel(uDotGoalP',qGoal',kC);
 73    
 74    nInitJoint = [qInit phiInit 0 qDotInit' 0 omegaInit];
 75    nGoalJoint = [qGoal 0 0 qDotGoal' 0 0];
 76    
 77    %Check that the initial and final positions are valid. If they are not
 78    %return failure and an empty path.
 79    if (validJointState(nInitJoint,jointLimits) && validJointState(nGoalJoint,jointLimits))
 80        success = true;
 81        %Run buildRRT.
 82        nInit = [1 0 0 nInitJoint];
 83        nGoal = [0 0 0 nGoalJoint];
 84        [T,pathJ] = buildRRT(nInit,nGoal,NUM_NODES,jointLimits,cartesianLimits,panHeight,HGAINS,NODE_SIZE,U,U_SIZE,dt,Dt,kC,ankleThreshold,exhaustive,threshold,goalSeedFreq,uBDot,legNum);
 85        %Transform path back to the Cartesian space.
 86        pathC = transformPath(pathJ,kC,TP2B);
 87    else
 88        success = false;
 89        pathC = [];
 90        pathJ = [];
 91        T = [];
 92    end 
 93
 94end

1%TRDH Generates the homogeneous transformation matrix A using the 
 2%Denavit-Hartenberg parameters theta, d, a and alpha.
 3%
 4%trDH.m
 5%author:    wreid
 6%date:      20150214
 7
 8function A = trDH(theta,d,a,alpha)
 9    A = [cos(theta) -sin(theta)*cos(alpha) sin(theta)*sin(alpha) a*cos(theta);...
10        sin(theta) cos(theta)*cos(alpha) -cos(theta)*sin(alpha) a*sin(theta);...
11        0 sin(alpha) cos(alpha) d;...
12        0 0 0 1];
13end

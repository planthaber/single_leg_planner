1function [xNew,transitionArray] = selectInput(xNear,xRand,U,dt,Dt,NODE_SIZE,U_SIZE,HGAINS,kC,ankleThreshold,jointLimits,uBDot,legNum)
 2%selectInput Selects the most appropriate control input.
 3%   A control input is selected from a set of control inputs, U. An input
 4%   is selected by applying each of the inputs to to state xNear, which
 5%   results in p candidate states, where p is the size of the input set.
 6%   The control input corresponding to candidate state that is closest to
 7%   x1 is returned as u.
 8
 9    %Initialize arrays to store the candidate new state data and the
10    %distances between each candidate state and the xNear state.
11    candStates = zeros(U_SIZE,NODE_SIZE);
12    candTransArrays = zeros(U_SIZE,(round(Dt/dt)+1)*10);
13    distance = zeros(1,U_SIZE);
14    
15    %UJoint = zeros(U_SIZE,3);
16    
17    %Transform the control inputs to joint space.
18    %for i = 1:U_SIZE
19        %gammaDotDot = (-betaDotDot*L3*cos(beta)+betaDot^2*L3*sin(beta)+gammaDot^2*L5*sin(zeta+gamma))/(L5*cos(zeta+gamma));
20        %gammaDotDot = getConstrainedGammaDotDot(kC,U(i,:),xNear(7:9),xNear(4:6));
21        %UJoint(i,:) = [U(i,:) gammaDotDot];
22    %end    
23
24    %Increment over the control vector. Generate a candidate state for each
25    %possible control input.
26    for i = 1:U_SIZE
27        
28        %Generate a candidate state using a fourth order Runge-Kutta 
29        %integration technique.
30        [candStates(i,:),candTransArrays(i,:)] = rk4(U(i,:),uBDot,dt,Dt,xNear,jointLimits,kC,legNum);
31        %U_check = sherpaTTFKAcc(U_joint(i,:)',xNear(7:9)',xNear(4:6)',kinematicConst)
32        %velCheck = sherpaTTFKVel(candStates(i,7:9)',candStates(i,4:6)',kinematicConst);
33       
34        %Calculate the distance between the candidate state and the random
35        %state.
36        hDiff = heuristicSingleLeg(candStates(i,:),xRand,HGAINS,jointLimits,kC);
37
38        %Apply the ankle constraint to penalize any candidate state that
39        %requires a change of ankle position greater than the allowed ankle
40        %movement in a single time step.
41        aGain = HGAINS(3);
42        
43        aDiff = angDiff(xNear(4),candStates(i,4));
44        ankleDiffMax = pi;
45        
46        if aDiff > ankleThreshold
47            aDiff = 1;
48        else
49            %aDiff = abs(aDiff/ankleDiffMax);
50            aDiff = 0;
51        end
52        
53        %Calculate a distance metric that includes the heurisitc distance
54        %as well as any penalty due to ankle movements.
55        distance(i) = (1-aGain)*hDiff + aGain*aDiff;
56        %distance(i) = hDiff;
57    end
58    
59    
60    [~,minIndex] = min(distance);
61    xNew = candStates(minIndex,:);
62    transitionArray = candTransArrays(minIndex,:);
63    
64    %velCheck = sherpaTTFKVel(xNew(7:9)',xNew(4:6)',kinematicConst)
65
66end

124function xDot = f(x,u,kC)
125    
126    alpha = x(1);
127    beta = x(2);
128    gamma = x(3);
129    
130    alphaDot = x(6);
131    betaDot = x(7);
132    gammaDot = x(8);
133    
134    alphaDotDot = u(1);
135    betaDotDot = u(2);
136    gammaDotDot = (-betaDotDot*kC.l3*cos(beta)+betaDot^2*kC.l3*sin(beta)+gammaDot^2*kC.l5*sin(kC.zeta+gamma))/(kC.l5*cos(kC.zeta+gamma));
137    
138    xDot = [alphaDot betaDot gammaDot 0 0 alphaDotDot betaDotDot gammaDotDot 0 0]; 
139    
140end

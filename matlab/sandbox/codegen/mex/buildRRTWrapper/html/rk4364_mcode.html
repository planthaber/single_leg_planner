  5function [xNew,transitionArray] = rk4(uIn,uBDot,dt,Dt,xInit,jointLimits,kC,legNum)
  6%rk4 Summary of this function goes here
  7%   Detailed explanation goes here
  8
  9    u = uIn;
 10
 11    numIterations = round(Dt/dt);
 12    xNew = zeros(1,length(xInit));
 13    xInit = xInit(4:end);
 14    xInitOrig = xInit;
 15    transitionArray = zeros(1,(numIterations+1)*10);
 16    transitionArray(1:10) = xInit;
 17        
 18    for i = 1:numIterations
 19
 20        xNew = numIntegrate(xInit,u,kC,dt);
 21        
 22        phiInit = xInit(4);
 23        
 24        alpha = xNew(1);
 25        beta = xNew(2);
 26        gamma = xNew(3);
 27        epsilon = xNew(5);
 28
 29        alphaDot = xNew(6);
 30        betaDot = xNew(7);
 31        gammaDot = xNew(8);
 32        phiDot = xNew(9);
 33        omega = xNew(10);
 34        
 35        alphaDotDot = u(1);
 36        betaDotDot = u(2);
 37        
 38        alphaMax = jointLimits(2,1);
 39        alphaMin = jointLimits(1,1);
 40        betaMax = jointLimits(2,2);
 41        betaMin = jointLimits(1,2);
 42        gammaMax = jointLimits(2,3);
 43        gammaMin = jointLimits(1,3);
 44        phiMax = jointLimits(2,4);
 45        phiMin = jointLimits(1,4);
 46        
 47        alphaDotMax = jointLimits(2,6);
 48        alphaDotMin = jointLimits(1,6);
 49        betaDotMax = jointLimits(2,7);
 50        betaDotMin = jointLimits(1,7);
 51        gammaDotMax = jointLimits(2,8);
 52        gammaDotMin = jointLimits(1,8);
 53        phiDotMax = jointLimits(2,9);
 54        phiDotMin = jointLimits(1,9);
 55        omegaMax = jointLimits(2,10);
 56        omegaMin = jointLimits(1,10);
 57        
 58        %Check pan angular position limits
 59        if alpha > alphaMax || alpha < alphaMin
 60            alpha = xInit(1);
 61            alphaDot = 0;
 62            alphaDotDot = 0;
 63        end
 64        
 65         %Check inner and outer leg angular position limits
 66        if beta > betaMax || beta < betaMin || gamma > gammaMax || gamma < gammaMin
 67            beta = xInit(2);
 68            gamma = xInit(3);
 69            betaDot = 0;
 70            gammaDot = 0;
 71            betaDotDot = 0;
 72        end
 73        
 74        %Check pan angular velocity limits
 75        if alphaDot > alphaDotMax || alphaDot < alphaDotMin
 76            alphaDot = xInit(6);
 77            alphaDotDot = 0;
 78        %else
 79        %    alphaDotDot = uIn(1);
 80        end
 81        
 82        %Check inner and outer leg angular velocity limits
 83        if betaDot > betaDotMax || betaDot < betaDotMin || gammaDot > gammaDotMax || gammaDot < gammaDotMin
 84            betaDot = xInit(7);
 85            gammaDot = xInit(8);
 86            betaDotDot = 0;
 87        %else
 88        %    betaDotDot = uIn(2);
 89        end
 90        
 91        [phi,omega] = getPhiAndOmega(uBDot,[alphaDot betaDot gammaDot 0],[alpha beta gamma 0],phiInit,kC,legNum);
 92        epsilon = epsilon + dt*omega;
 93        
 94%         %Check if phi is above threshold, if so then stop the leg and drive
 95%         %the steering joint until it is in the correct orientation.
 96%         if abs(phi-phiInit)/dt > phiDotMax
 97%           if phi > phiInit
 98%             phiDot = phiDotMax;
 99%           else
100%             phiDot = phiDotMin;  
101%           end
102%           alphaDotDot = 0;
103%           betaDotDot = 0;
104%           alphaDot = 0;
105%           betaDot = 0;
106%           gammaDot = 0;
107%           omega = 0;
108%           alpha = xInitOrig(1);
109%           beta = xInitOrig(2);
110%           gamma = xInitOrig(3);
111%         end
112        
113        u = [alphaDotDot betaDotDot];
114        
115        xNew = [alpha beta gamma phi epsilon alphaDot betaDot gammaDot phiDot omega];
116        
117        xInit = xNew;
118        transitionArray((10*i)+1:(10*(i+1))) = xNew;
119    end
120    xNew = [zeros(1,3) xNew];
121     
122end

120function pathC = transformPath(pathJ,kC,TP2B)
121    [pathH,~] = size(pathJ);
122    pathC = zeros(pathH,9);
123    dist2Go = 0;
124    for i = 1:pathH
125        uP = sherpaTTFK(pathJ(i,2:4),kC);
126        uPDot = sherpaTTFKVel(pathJ(i,7:9),pathJ(i,2:4),kC);
127        uB = TP2B(1:3,1:3)*uP' + TP2B(1:3,4);
128        uBDot = TP2B(1:3,1:3)*uPDot;
129        if i ~= 1
130            dist2Go = dist2Go + norm(uB'-pathC(i-1,3:5));
131        end
132        pathC(i,:) = [pathJ(i,1) dist2Go uB' uBDot' false];
133    end
134end

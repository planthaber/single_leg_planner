1%BUILDRRTWRAPPER This function acts as a wrapper for the buildRRT function.
  2%Code generation for the singleLegPlanner is performed using this function
  3%as an entry point.
  4%  
  5%Inputs:
  6%-nInitCartesianB: the
  7%-nGoalCartesianB:
  8%-jointLimits:
  9%-bodyHeight:
 10%-U:
 11%-dt:
 12%-Dt:
 13%-kC:
 14%-threshold:
 15%-legNum:
 16%
 17%Outputs:
 18%-T:
 19%-pathC:
 20%-pathJ:
 21%-success:
 22%
 23%buildRRTWrapper.m
 24%author: wreid
 25%date: 20150502
 26
 27function [T,pathC,pathJ,success] = buildRRTWrapper(nInitCartesianB,nGoalCartesianB,jointLimits,bodyHeight,U,dt,Dt,kC,threshold,legNum,uBDot)
 28
 29    persistent NUM_NODES 
 30    persistent HGAINS
 31    persistent NODE_SIZE 
 32    persistent U_SIZE 
 33    persistent ankleThreshold 
 34    persistent exhaustive 
 35    persistent goalSeedFreq 
 36    persistent cartesianLimits
 37    
 38    if isempty(NUM_NODES)
 39        NUM_NODES = int32(1000);
 40    end
 41    if isempty(HGAINS)
 42        HGAINS = [1 0 0.5];
 43    end
 44    if isempty(NODE_SIZE)
 45        NODE_SIZE = int32(11);
 46    end
 47    if isempty(U_SIZE)
 48        U_SIZE = int32(5);
 49    end
 50    if isempty(ankleThreshold)
 51        ankleThreshold = pi/8;
 52    end
 53    if isempty(exhaustive)
 54        exhaustive = false;
 55    end
 56    if isempty(goalSeedFreq)
 57        goalSeedFreq = int32(20);
 58    end
 59    
 60    if isempty(cartesianLimits)
 61        cartesianLimits = zeros(1,4);
 62        betaMin = jointLimits(1,2);
 63        betaMax = jointLimits(2,2);
 64        gammaMin = jointLimits(1,3);
 65        gammaMax = jointLimits(2,3);
 66        u1 = sherpaTTFK([0 betaMin gammaMin],kC);
 67        u2 = sherpaTTFK([0 betaMax gammaMax],kC);
 68        u3 = sherpaTTFK([0 betaMin gammaMax],kC);
 69        u4 = sherpaTTFK([0 betaMax gammaMin],kC);
 70        cartesianLimits(1) = u1(3);
 71        cartesianLimits(2) = u2(3);
 72        cartesianLimits(3) = u3(3);
 73        cartesianLimits(4) = u4(3);
 74    end
 75
 76    
 77    panHeight  = getPanHeight(bodyHeight,kC);
 78
 79    %Transform the nInitCartesianB and nGoalCartesianB variables from the body coordinate frame
 80    %to the pan coordinate frame.
 81    TP2B = trP2B(kC,legNum);
 82    TB2P = trInv(TP2B); %inv(TP2B);%
 83    uInitP = TB2P(1:3,1:3)*nInitCartesianB(1:3)' + TB2P(1:3,4);
 84    uDotInitP = TB2P(1:3,1:3)*nInitCartesianB(4:6)';
 85    uGoalP = TB2P(1:3,1:3)*nGoalCartesianB(1:3)' + TB2P(1:3,4);
 86    uDotGoalP = TB2P(1:3,1:3)*nGoalCartesianB(4:6)';
 87    
 88    %Transform the Cartesian goal and final positions in the pan coordinate
 89    %frame to the joint space.
 90    qInit = sherpaTTIK(uInitP,kC,jointLimits);
 91    qGoal = sherpaTTIK(uGoalP,kC,jointLimits);
 92    qDotInit = sherpaTTIKVel(uDotInitP',qInit',kC);
 93    qDotGoal = sherpaTTIKVel(uDotGoalP',qGoal',kC);
 94    
 95    nInitJoint = [qInit qDotInit'];
 96    nGoalJoint = [qGoal qDotGoal'];
 97    
 98    %Check that the initial and final positions are valid. If they are not
 99    %return failure and an empty path.
100    if (validJointState(nInitJoint,jointLimits) && validJointState(nGoalJoint,jointLimits))
101        success = true;
102        %Run buildRRT.
103        nInit = [0 0 0 nInitJoint 0 0];
104        nGoal = [0 0 0 nGoalJoint 0 0];
105        [T,pathJ] = buildRRT(nInit,nGoal,NUM_NODES,jointLimits,cartesianLimits,panHeight,HGAINS,NODE_SIZE,U,U_SIZE,dt,Dt,kC,ankleThreshold,exhaustive,threshold,goalSeedFreq,uBDot,legNum);
106        %Transform path back to the Cartesian space.
107        [pathC,pathJ] = transformPath(pathJ,NODE_SIZE,kC,dt,Dt,TP2B);
108        pathC = [0 nInitCartesianB; pathC];
109        pathJ = [0 qInit qDotInit'; pathJ];
110    else
111        success = false;
112        pathC = [];
113        pathJ = [];
114        T = [];
115    end 
116
117end

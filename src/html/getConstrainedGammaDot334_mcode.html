1%GETCONSTRAINEDGAMMADOT This function calculates the velocity of
 2%gamma given a pan height constraint and an independent beta angle.
 3%
 4%getConstrainedGammaDot.m
 5%author: wreid
 6%date: 20150224
 7
 8function gammaDot = getConstrainedGammaDot(kC,qDot,q)
 9
10    betaDot = qDot(2);
11    beta = q(2);
12    gamma = q(3);
13
14    gammaDot = -betaDot*kC.l3*cos(beta)/(kC.l5*cos(kC.zeta+gamma));
15
16end

106function pathC = transformPath(pathJ,kC,TP2B)
107    [pathH,~] = size(pathJ);
108    pathC = zeros(pathH,8);
109    dist2Go = 0;
110    for i = 1:pathH
111        uP = sherpaTTFK(pathJ(i,2:4),kC);
112        uPDot = sherpaTTFKVel(pathJ(i,7:9),pathJ(i,2:4),kC);
113        uB = TP2B(1:3,1:3)*uP' + TP2B(1:3,4);
114        uBDot = TP2B(1:3,1:3)*uPDot;
115        if i ~= 1
116            dist2Go = dist2Go + norm(uB'-pathC(i-1,3:5));
117        end
118        pathC(i,:) = [pathJ(i,1) dist2Go uB' uBDot'];
119    end
120end

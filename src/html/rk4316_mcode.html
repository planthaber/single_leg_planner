 5function [xNew,transitionArray] = rk4(u,dt,Dt,xInit,jointLimits)
 6%rk4 Summary of this function goes here
 7%   Detailed explanation goes here
 8
 9    numIterations = round(Dt/dt);
10    xNew = zeros(1,length(xInit));
11    xInit = xInit(4:end-2);
12    %xInitOrig = xInit;
13    transitionArray = zeros(1,(numIterations+1)*6);
14    transitionArray(1:6) = xInit;
15        
16    for i = 1:numIterations
17        
18        k1 = f(xInit,u); 
19        k2 = f(xInit+dt/2*k1,u);    
20        k3 = f(xInit+dt/2*k2,u);
21        k4 = f(xInit+dt/2*k3,u);
22        xNew = xInit + dt/6*(k1 + 2*k2 + 2*k3 + k4);
23        
24        %Check pan angular position limits
25        if xNew(1) > jointLimits(2,1) || xNew(1) < jointLimits(1,1)
26            xNew(1) = xInit(1);
27            xNew(4) = 0;
28            u(1) = 0;
29        end
30        
31         %Check inner and outer leg angular position limits
32        if xNew(2) > jointLimits(2,2) || xNew(2) < jointLimits(1,2) || xNew(3) > jointLimits(2,3) || xNew(3) < jointLimits(1,3)
33            xNew(2) = xInit(2);
34            xNew(3) = xInit(3);
35            xNew(5) = 0;
36            xNew(6) = 0;
37            u(2) = 0;
38            u(3) = 0;
39        end
40        
41        %Check pan angular velocity limits
42        if xNew(4) > jointLimits(2,4) || xNew(4) < jointLimits(1,4)
43            xNew(4) = xInit(4);
44            u(1) = 0;
45        end
46        
47        %Check inner and outer leg angular velocity limits
48        if xNew(5) > jointLimits(2,5) || xNew(5) < jointLimits(1,5) || xNew(6) > jointLimits(2,6) || xNew(6) < jointLimits(1,6)
49            xNew(5) = xInit(5);
50            xNew(6) = xInit(6);
51            u(2) = 0;
52            u(3) = 0;
53        end
54        
55        xInit = xNew;
56        transitionArray((6*i)+1:(6*(i+1))) = xNew;
57    end
58    
59    %xInit = [zeros(1,3) xInitOrig 0 0];
60    xNew = [zeros(1,3) xNew 0 0];
61     
62end

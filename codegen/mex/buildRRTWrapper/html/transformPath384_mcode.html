118function pathC = transformPath(pathJ,kC,TP2B)
119    [pathH,~] = size(pathJ);
120    pathC = zeros(pathH,9);
121    dist2Go = 0;
122    for i = 1:pathH
123        uP = sherpaTTFK(pathJ(i,2:4),kC);
124        uPDot = sherpaTTFKVel(pathJ(i,7:9),pathJ(i,2:4),kC);
125        uB = TP2B(1:3,1:3)*uP' + TP2B(1:3,4);
126        uBDot = TP2B(1:3,1:3)*uPDot;
127        if i ~= 1
128            dist2Go = dist2Go + norm(uB'-pathC(i-1,3:5));
129        end
130        pathC(i,:) = [pathJ(i,1) dist2Go uB' uBDot' false];
131    end
132end
